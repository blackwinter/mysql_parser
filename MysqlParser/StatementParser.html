def parse(input)
  case input
    when USE_RE           then yield :use,    $1
    when CREATE_TABLE_RE  then yield :create, $1
    when TABLE_COLUMN_RE  then yield :column, $1
    when FINISH_TABLE_RE  then yield :table
    when INSERT_VALUES_RE then yield :insert, $1, $2, $3
  end
end
